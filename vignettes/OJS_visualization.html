bubbles_pq_ojs <- function(physeq,
                       rank_label="Taxa",
                       tooltip = "",
                       rank_color = "Family",
                       label_color = "grey20",
                       label_size= 8,
                       width = NULL,
                       height = NULL,
                       log1ptransform = FALSE,
                       defaultWidth = 600,
                       defaultHeight = 600,
                       min_nb_seq_label = 0) {

  library(htmlwidgets)
  if(rank_label=="Taxa") {
    label <- taxa_names(physeq)
  } else {
    label <- physeq@tax_table[, rank_label]
  }

  if(min_nb_seq_label > 0 && min(taxa_sums(physeq))<min_nb_seq_label) {
    label[taxa_sums(physeq)<min_nb_seq_label] <- NA
  }


  df <- data.frame(
    value = taxa_sums(physeq),
    label = label,
    tooltip = tooltip,
    color = fac2col(physeq@tax_table[, rank_color]),
    textColor = label_color
  )

  if (log1ptransform) {
    df$value <- log1p(df$value)
  }

  return(df)

}

chart = {
  // Specify the dimensions of the chart.
  const width = 928;
  const height = width;
  const margin = 1; // to avoid clipping the root circle stroke
  const name = d => d.label.split(".").pop(); // "Strings" of "flare.util.Strings"
  const group = d => d.label.split(".")[1]; // "util" of "flare.util.Strings"
  const names = d => name(d).split(/(?=[A-Z][a-z])|\s+/g); // ["Legend", "Item"] of "flare.vis.legend.LegendItems"

  // Specify the number format for values.
  const format = d3.format(",d");

  // Create a categorical color scale.
  const color = d3.scaleOrdinal(d3.schemeTableau10);

  // Create the pack layout.
  const pack = d3.pack()
      .size([width - margin * 2, height - margin * 2])
      .padding(3);

  // Compute the hierarchy from the (flat) data; expose the values
  // for each node; lastly apply the pack layout.
  const root = pack(d3.hierarchy({children: data})
      .sum(d => d.value));

  // Create the SVG container.
  const svg = d3.create("svg")
      .attr("width", width)
      .attr("height", height)
      .attr("viewBox", [-margin, -margin, width, height])
      .attr("style", "max-width: 100%; height: auto; font: 10px sans-serif;")
      .attr("text-anchor", "middle");

  // Place each (leaf) node according to the layoutâs x and y values.
  const node = svg.append("g")
    .selectAll()
    .data(root.leaves())
    .join("g")
      .attr("transform", d => `translate(${d.x},${d.y})`);

  // Add a title.
  node.append("title")
      .text(d => `${d.data.id}\n${format(d.value)}`);

  // Add a filled circle.
  node.append("circle")
      .attr("fill-opacity", 0.7)
      .attr("fill", d => color(group(d.data)))
      .attr("r", d => d.r);

  // Add a label.
  const text = node.append("text")
      .attr("clip-path", d => `circle(${d.r})`);

  // Add a tspan for each CamelCase-separated word.
  text.selectAll()
    .data(d => names(d.data))
    .join("tspan")
      .attr("x", 0)
      .attr("y", (d, i, nodes) => `${i - nodes.length / 2 + 0.35}em`)
      .text(d => d);

  // Add a tspan for the nodeâs value.
  text.append("tspan")
      .attr("x", 0)
      .attr("y", d => `${names(d.data).length / 2 + 0.35}em`)
      .attr("fill-opacity", 0.7)
      .text(d => format(d.value));

  return Object.assign(svg.node(), {scales: {color}});
